import { Step, Steps } from 'fumadocs-ui/components/steps';
import { CodeTabs } from '@/components/docs/code-tabs';

<Steps>
<Step>

### Initialize anime-ui

Run the `init` command to create a new project with anime-ui or setup an existing one.

<CodeTabs
  codes={{
    pnpm: `pnpm dlx anime-ui-cli@beta init`,
    npm: `npx anime-ui-cli@beta init`,
    yarn: `npx anime-ui-cli@beta init`,
    bun: `bunx anime-ui-cli@beta init`,
  }}
/>

</Step>
<Step>

### Add Components

Run the `add` command to add a component to your project.

<CodeTabs
  codes={{
    pnpm: `pnpm dlx anime-ui-cli@beta add block-text`,
    npm: `npx anime-ui-cli@beta add block-text`,
    yarn: `npx anime-ui-cli@beta add block-text`,
    bun: `bunx anime-ui-cli@beta add block-text`,
  }}
/>

</Step>
<Step>

### Import Components

Import the component to your project.

```tsx
import { Button } from '@/components/anime-ui/button';

export default function Home() {
  return (
    <div>
      <Button />
    </div>
  );
}
```

</Step>
</Steps>
